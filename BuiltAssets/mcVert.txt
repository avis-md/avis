#version 330 core

uniform ivec3 sz;

out vec3 v2g_uvw;

void main() {
	int iz = gl_VertexID / (sz.x * sz.y);
	int miz = int(mod(gl_VertexID, sz.x * sz.y));
	int iy = miz / sz.x;
	int ix = int(mod(miz, sz.x));
	v2g_uvw = vec3((ix+0.5)/sz.x, (iy+0.5)/sz.y, (iz+0.5)/sz.z);
	gl_Position = vec4(0,0,0,0);
}
