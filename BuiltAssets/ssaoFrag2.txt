#version 330

//in vec2 uv;

uniform sampler2D tex1;
uniform sampler2D tex2;
uniform float val;
uniform vec2 screenSize;

out vec4 fragCol;

void main () {
	vec2 uv = gl_FragCoord.xy / screenSize;
	//fragCol = texture(tex1, uv) - ((1-texture(tex2, uv).r) * val);
	fragCol = mix(texture(tex1, uv), vec4(0,0,0,0), (1-texture(tex2, uv).r) * val);
}
